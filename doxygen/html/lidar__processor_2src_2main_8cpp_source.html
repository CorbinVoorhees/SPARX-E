8/* Summary:
9Converts /unilidar/cloud PointCloud2 data into XYZ points, stores them in a class, then processes them into:
101. filtered obstacle-candidate points
112. KD-tree
123. obstacle clusters
134. Gaussian obstacle cost model
14
15More assumptions
16- The LiDAR topic is /unilidar/cloud
17- The topic type is sensor_msgs/msg/PointCloud2
18- PCL is available (point cloud library)
19- Rover body frame convention:
20 x = forward
21 y = left
22 z = up
23- LiDAR points initially arrive in the LiDAR frame
24- LiDAR mounting error is modeled as a yaw rotation about z then 180 deg roll about x
25- Points close to ground are ignored as non-obstacles
26- Obstacles are found by clustering remaining points
27- Each obstacle cluster becomes one Gaussian cost source
28*/

196 void estimation_callback(const nav_msgs::msg::Odometry::SharedPtr est_msg)
197 {
198 std::lock_guard<std::mutex> lg(estimation_mutex);
199 // get current position, orientation, velocity.
200 auto &p = est_msg->pose.pose;
201 pos[0] = p.position.x;
202 pos[1] = p.position.y;
203 pos[2] = p.position.z;
204
205 // get the orientation
206 Eigen::Quaterniond quat(p.orientation.w, p.orientation.x, p.orientation.y, p.orientation.z);
207
208 static constexpr double MY_PI = 3.141592653589793238462643383279502884;
209 auto R = quat.toRotationMatrix();
210 double roll = std::atan2(R(2, 1), R(2, 2));
211 double s = -R(2, 0);
212 double pitch = (std::abs(s) >= 1.0) ? std::copysign(MY_PI / 2.0, s) : std::asin(s);
213 double yaw = std::atan2(R(1, 0), R(0, 0));
214
215 orientation[0] = roll;
216 orientation[1] = pitch;
217 orientation[2] = yaw;
218
219 // get velos
220 auto &t = est_msg->twist.twist;
221 velocity[0] = t.linear.x;
222 velocity[1] = t.linear.y;
223 velocity[2] = t.linear.z;
224 }

226 // CALLBACK - raw ROS data to XYZ
227 void pointCloudCallback(const sensor_msgs::msg::PointCloud2::SharedPtr msg)
228 {
229 raw_cloud_lidar_->clear();
230 pcl::fromROSMsg(*msg, *raw_cloud_lidar_);
231
232 // Process transformations outside the lock to minimize blocking
233 pcl::PointCloud<pcl::PointXYZ> temp_cloud;
234
235 for (const auto& p_lidar : raw_cloud_lidar_->points) {
236 Eigen::Vector3f p_L(p_lidar.x, p_lidar.y, p_lidar.z);
237 Eigen::Vector3f p_B = R_body_lidar_ * p_L;
238
239 pcl::PointXYZ p_body;
240 p_body.x = p_B.x();
241 p_body.y = p_B.y();
242 p_body.z = p_B.z();
243
244 temp_cloud.points.push_back(p_body);
245 }
246
247 // Lock mutex and append the newly transformed points to the accumulator
248 {
249 std::lock_guard<std::mutex> lock(cloud_mutex_);
250 *accumulated_cloud_ += temp_cloud; // PCL supports direct point cloud concatenation
251 accumulated_cloud_->width = accumulated_cloud_->points.size();
252 accumulated_cloud_->height = 1;
253 accumulated_cloud_->is_dense = false;
254 }

317 {
318 Eigen::Matrix3f R_yaw;
319 R_yaw << std::cos(mount_offset_angle_rad_), -std::sin(mount_offset_angle_rad_), 0.0f,
320 std::sin(mount_offset_angle_rad_), std::cos(mount_offset_angle_rad_), 0.0f,

333 void filterPoints()
334 {
335 filtered_cloud_->clear();
336
337 float projector_height = 4.0f;
338 float rover_height = 0.245f;
339 float ground_z = -rover_height;
340
341 for (const auto& p : cloud_body_->points) {
342 if (!std::isfinite(p.x) || !std::isfinite(p.y) || !std::isfinite(p.z)) {
343 continue;
344 }
345
346 if (p.x < min_x_ || p.x > max_x_) continue;
347 if (std::abs(p.y) > max_abs_y_) continue;
348
349 if (p.z <= (ground_z + ground_threshold_)) {
350 continue;
351 }
352
353 if (p.z > projector_height) continue;
354
355 if (p.z > 1.0f) {
356 RCLCPP_WARN(this->get_logger(), "High point detected: z=%.2f", p.z);
357 }
358
359 filtered_cloud_->points.push_back(p);
360 }
361
362 filtered_cloud_->width = filtered_cloud_->points.size();
363 filtered_cloud_->height = 1;
364 filtered_cloud_->is_dense = false;
365 }

378 void computeObstacleClusters()
379 {
380 if (filtered_cloud_->empty()) {
381 std::lock_guard<std::mutex> lock(obstacles_mutex_);
382 obstacles_.clear();
383 return;
384 }
385
386 std::vector<pcl::PointIndices> cluster_indices;
387 pcl::EuclideanClusterExtraction<pcl::PointXYZ> ec;
388 ec.setClusterTolerance(cluster_tolerance_);
389 ec.setMinClusterSize(min_cluster_size_);
390 ec.setMaxClusterSize(max_cluster_size_);
391 ec.setSearchMethod(kd_tree_);
392 ec.setInputCloud(filtered_cloud_);
393 ec.extract(cluster_indices);
394
395 // Compute temporarily, so we hold the lock for as little time as possible
396 std::vector<Obstacle> new_obstacles;
397
398 for (const auto& cluster : cluster_indices) {
399 Obstacle obs;
400 obs.center_xy = Eigen::Vector2f(0.0f, 0.0f);
401 obs.height = -std::numeric_limits<float>::infinity();
402 obs.num_points = static_cast<int>(cluster.indices.size());
403
404 for (int idx : cluster.indices) {
405 const auto& p = filtered_cloud_->points[idx];
406
407 obs.center_xy.x() += p.x;
408 obs.center_xy.y() += p.y;
409
410 if (p.z > obs.height) {
411 obs.height = p.z;
412 }
413 }
414
415 obs.center_xy /= static_cast<float>(obs.num_points);
416
417 Eigen::Matrix2f cov = Eigen::Matrix2f::Zero();
418
419 for (int idx : cluster.indices) {
420 const auto& p = filtered_cloud_->points[idx];
421
422 Eigen::Vector2f d;
423 d << p.x - obs.center_xy.x(),
424 p.y - obs.center_xy.y();
425
426 cov += d * d.transpose();
427 }
428
429 cov /= (static_cast<float>(obs.num_points) - 1);
430
431 float min_sigma = 0.25f;
432 cov += Eigen::Matrix2f::Identity() * min_sigma * min_sigma;
433
434 obs.covariance = cov;
435 obs.inv_covariance = cov.inverse();
436 obs.amplitude = base_amplitude_ * (1.0f + 0.10f * obs.height);
437
438 new_obstacles.push_back(obs);
439 }
440
441 // Lock mutex, update real container safely
442 {
443 std::lock_guard<std::mutex> lock(obstacles_mutex_);
444 obstacles_ = std::move(new_obstacles);
445 }
446 }

448 void computeGaussianCosts()
449 {
450 std::lock_guard<std::mutex> lock(obstacles_mutex_);
451 for (auto& obs : obstacles_) {
452 obs.amplitude = base_amplitude_ * (1.0f + 0.10f * obs.height);
453 }

456 float gaussianCostAtPoint(float x, float y) const
457 {
458 float total_cost = 0.0f;
459
460 Eigen::Vector2f p;
461 p << x, y;
462
463 for (const auto& obs : obstacles_) {
464 Eigen::Vector2f d = p - obs.center_xy;
465 float exponent = -0.5f * d.transpose() * obs.inv_covariance * d;
466 float cost = obs.amplitude * std::exp(exponent);
467 total_cost += cost;
468 }
469
470 return total_cost;
471 }
472
473 float pathCostStraightLine(float heading_rad, float path_length) const
474 {
475 int num_samples = 50;
476 float total_cost = 0.0f;
477
478 for (int i = 1; i <= num_samples; i++) {
479 float s = path_length * static_cast<float>(i) / static_cast<float>(num_samples);
480 float x = s * std::cos(heading_rad);
481 float y = s * std::sin(heading_rad);
482
483 total_cost += gaussianCostAtPoint(x, y);
484 }
485
486 return total_cost;
487 }
