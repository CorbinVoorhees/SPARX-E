41def clean(df):
42 if 'time' not in df.columns:
43 raise KeyError("Gains CSV missing 'time' column.")
44
45 all_groups = [g for _, g in GROUPS] + [g for _, g, _ in INNOV_GROUPS]
46 cols = ['time'] + [c for group in all_groups for c in group if c in df.columns]
47 df = df.copy()
48 for c in cols:
49 df[c] = pd.to_numeric(df[c], errors='coerce')
50
51 df = df.dropna(subset=['time']).sort_values('time')

53 if len(df) < 2:
54 raise RuntimeError('Not enough gain rows to plot. Run nav_node until sparxe_gains.csv has data rows.')

58def plot_group(ax, t, df, title, cols, ylabel='|K| row norm'):
59 plotted = False
60 for c in cols:
61 if c not in df.columns:
62 continue
63 y = df[c].to_numpy(dtype=float)
64 if np.all(~np.isfinite(y)):
65 continue
66 ax.plot(t, y, linewidth=1.2, label=c)

69 ax.set_title(title, fontsize=11)
70 ax.set_ylabel(ylabel)
