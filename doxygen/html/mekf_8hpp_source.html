26 void apply_error(const Eigen::VectorXd &err) override {
27 const Eigen::Vector3d dtheta = err.segment<3>(0);
28 const Eigen::Vector4d o = q();
29 Eigen::Quaterniond q_curr(o(0), o(1), o(2), o(3));
30 Eigen::Quaterniond dq(1.0, 0.5 * dtheta.x(), 0.5 * dtheta.y(),
31 0.5 * dtheta.z());
32 Eigen::Quaterniond q_new = (q_curr * dq).normalized();
33 if (q_new.w() < 0.0)
34 q_new.coeffs() *= -1.0;
35 q() << q_new.w(), q_new.x(), q_new.y(), q_new.z();

38 void predict(double dt) override {
39 if (!gyro_snapshot.fresh)
40 return;
41
42 const Eigen::Vector3d w_corr = gyro_snapshot.z - bg0;
43
44 const Eigen::Vector4d o = q();
45 Eigen::Quaterniond prior_q(o(0), o(1), o(2), o(3));
46 Eigen::Quaterniond omega(0.0, w_corr.x(), w_corr.y(), w_corr.z());
47 Eigen::Quaterniond q_dot;
48 q_dot.coeffs() = 0.5 * (prior_q * omega).coeffs();
49 Eigen::Quaterniond q_pred;
50 q_pred.coeffs() = prior_q.coeffs() + q_dot.coeffs() * dt;
51 q_pred.normalize();
52 if (q_pred.w() < 0.0)
53 q_pred.coeffs() *= -1.0;
54 q() << q_pred.w(), q_pred.x(), q_pred.y(), q_pred.z();
55
56 const Eigen::Matrix3d F = -skew(w_corr);
57 const Eigen::Matrix3d Qk = Qw * dt;
58 const Eigen::Matrix3d Phi = Eigen::Matrix3d::Identity() + F * dt;
59 P = Phi * P * Phi.transpose() + Qk;
60 }
