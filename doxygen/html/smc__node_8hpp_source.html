39 // Jacobian function --> input: geometry (Rw/B), slip efficiencies, orientation
40 static Eigen::Matrix<double, 3, 2> b_jacobian(double r_w, double B, double mu_R, double mu_L, Eigen::Vector3d &orientation)
41 {
42
43 // hardcode the jacobian matrix to be returned
44 Eigen::Matrix<double, 3, 2> dBdu;
45
46 // set all values to zero (they're random)
47 dBdu.setZero();
48
49 // unpack all of the orientation angles
50 double roll = orientation[0]; // rad
51 double pitch = orientation[1]; // rad
52 double yaw = orientation[2]; // rad
53
54 // actually input values for jacobian matrix (derivative of X_dot WRT control)
55 dBdu << (r_w / 2) * mu_R * cos(pitch) * cos(yaw), (r_w / 2) * mu_L * cos(pitch) * cos(yaw),
56 (r_w / 2) * mu_R * cos(pitch) * sin(yaw), (r_w / 2) * mu_L * cos(pitch) * sin(yaw),
57 (r_w / B) * mu_R * cos(roll) * cos(pitch), (-r_w / B) * mu_L * cos(roll) * cos(pitch);
58
59 // return
60 return dBdu;
61 }

63 // function to compute B matrix evaluated at current state (and previous control input)
64 static Eigen::Vector3d B_Xu(double r_w, double B, double mu_R, double mu_L, Eigen::Vector3d &orientation, Eigen::Vector4d &input)
65 {
66
67 // hardcode the B matrix to be returned
68 Eigen::Vector3d B_vec;
69
70 // set all values to zero (they're random)
71 B_vec.setZero();
72
73 // unpack all of the orientation angles
74 double roll = orientation[0]; // rad
75 double pitch = orientation[1]; // rad
76 double yaw = orientation[2]; // rad
77
78 // unpack control inputs
79 double wR = input[0]; // rad/s
80 double wL = input[1]; // rad/s
81 // input[2], input[3] (dwR, dwL) currently unused in the dynamics
82
83 // actually input values for B matrix
84 B_vec << (r_w / 2) * (mu_R * wR + mu_L * wL) * cos(pitch) * cos(yaw), // m/s
85 (r_w / 2) * (mu_R * wR + mu_L * wL) * cos(pitch) * sin(yaw), // m/s
86 (r_w / B) * (mu_R * wR - mu_L * wL) * cos(roll) * cos(pitch); // rad/s
87
88 // return
89 return B_vec;
90 }

102 // CALLER MUST HOLD state_mutex.
103 // Returns the control increment du = [d_wR, d_wL] from the SMC law.
104 Eigen::Vector2d compute_control_increment(double r_w, double B, double mu_r, double mu_l)
105 {
106 Eigen::Vector3d eulers(this->latest_state_estimate[2],
107 this->latest_state_estimate[3],
108 this->latest_state_estimate[4]);
109
110 // matrix inversions -- materialized to a concrete type so we don't
111 // hold a lazy expression that references a destroyed temporary
112 Eigen::Matrix<double, 2, 3> dbdx_pseudo_inv =
113 b_jacobian(r_w, B, mu_r, mu_l, eulers)
114 .completeOrthogonalDecomposition()
115 .pseudoInverse();
116
117 Eigen::Vector3d B_prev = B_Xu(r_w, B, mu_r, mu_l, eulers, this->control);
118
119 // tracking-relevant state slice: [x, y, yaw]
120 Eigen::Vector3d x_track(this->latest_state_estimate[0],
121 this->latest_state_estimate[1],
122 this->latest_state_estimate[4]);
123 Eigen::Vector3d s = this->latest_state_desired.head<3>() - x_track;
124 Eigen::Vector3d xdot_d = this->latest_state_desired.tail<3>();
125
126 const double epsilon = 7.0;
127 Eigen::Vector3d K(0.1, 0.1, 2.0);
128
129 Eigen::Vector3d Ks = (K.array() * comp_tanh(s, epsilon).array()).matrix();
130 Eigen::Vector3d residual = -Ks - xdot_d - B_prev;
131
132 return dbdx_pseudo_inv * residual;
133 }

191 // function that runs when nav node gives us state estimation
192 void nav_callback(nav_msgs::msg::Odometry odom)
193 {
194 // extract state estimate from nav callback (do math outside the lock)
195 auto position = odom.pose.pose.position;
196 auto orientation = odom.pose.pose.orientation;
197
198 const double MY_PI = 3.141592653589;
199
200 Eigen::Quaterniond quat(orientation.w, orientation.x, orientation.y, orientation.z);
201 quat.normalize();
202 auto R = quat.toRotationMatrix();
203 double roll = std::atan2(R(2, 1), R(2, 2));
204 double s = -R(2, 0);
205 double pitch = (std::abs(s) >= 1.0) ? std::copysign(MY_PI / 2.0, s) : std::asin(s);
206 double yaw = std::atan2(R(1, 0), R(0, 0));
207
208 std::lock_guard<std::mutex> lk(this->state_mutex);
209 this->latest_state_estimate << position.x, position.y, roll, pitch, yaw;
210 }
