2"""

82def parse_mac_list(value: str) -> list[int]:
83 """Accept '0x1,0x2', '[0x1, 0x2]', or '1'."""
84 value = value.strip()
85 if value.startswith("["):
86 parsed = ast.literal_eval(value)
87 if not isinstance(parsed, list):
88 raise ValueError("MAC list must be a list")
89 return [parse_int_auto(x) for x in parsed]
90 return [parse_int_auto(x) for x in value.split(",") if x.strip()]

111class RangeStore:
112 def __init__(self, stale_ms: int = 600, alpha: float = 0.35) -> None:
113 self._lock = threading.Lock()
114 self._latest: dict[str, RangeSample] = {}
115 self._prev_range_m: dict[str, float] = {}
116 self._prev_time_ns: dict[str, int] = {}
117 self._smoothed_rate: dict[str, float] = {}
118 self.stale_ms = stale_ms
119 self.alpha = alpha
120
121 def update_from_ranging_data(self, data) -> None:
122 now_ns = time.monotonic_ns()
123 with self._lock:
124 for meas in getattr(data, "meas", []):
125 peer = getattr(meas, "mac_add", "unknown")
126 status_obj = getattr(meas, "status", None)
127 status_name = getattr(status_obj, "name", str(status_obj))
128 good = bool(status_obj == Status.Ok)
129
130 distance_cm = getattr(meas, "distance", None)
131 range_cm = int(round(float(distance_cm))) if distance_cm is not None else None
132 range_m = (range_cm / 100.0) if range_cm is not None else None
133
134 rate_mps: Optional[float] = None
135 rate_valid = False
136 if good and range_m is not None and peer in self._prev_range_m:
137 dt_s = (now_ns - self._prev_time_ns[peer]) / 1_000_000_000.0
138 if 0.02 <= dt_s <= 2.0:
139 raw_rate = (range_m - self._prev_range_m[peer]) / dt_s
140 prev_rate = self._smoothed_rate.get(peer, raw_rate)
141 rate_mps = self.alpha * raw_rate + (1.0 - self.alpha) * prev_rate
142 self._smoothed_rate[peer] = rate_mps
143 rate_valid = True
144
145 if good and range_m is not None:
146 self._prev_range_m[peer] = range_m
147 self._prev_time_ns[peer] = now_ns
148
149 sample = RangeSample(
150 ok=good,
151 peer=peer,
152 seq=int(getattr(data, "idx", -1)),
153 session_handle=int(getattr(data, "session_handle", -1)),
154 range_cm=range_cm,
155 range_m=range_m,
156 range_rate_mps=rate_mps,
157 range_rate_valid=rate_valid,
158 rate_source="derived_delta_range_over_delta_time",
159 age_ms=0,
160 status=status_name,
161 nlos=bool(getattr(meas, "nlos", False)) if hasattr(meas, "nlos") else None,
162 rssi_dbm=float(getattr(meas, "rssi")) if hasattr(meas, "rssi") else None,
163 host_time_ns=now_ns,
164 )
165 self._latest[peer] = sample
166
167 def get(self, peer: Optional[str] = None) -> RangeSample | None:
168 now_ns = time.monotonic_ns()
169 with self._lock:
170 if not self._latest:
171 return None
172 if peer is None:
173 # Return newest sample.
174 sample = max(self._latest.values(), key=lambda x: x.host_time_ns)
175 else:
176 sample = self._latest.get(peer)
177 if sample is None:
178 return None
179 age_ms = int((now_ns - sample.host_time_ns) / 1_000_000.0)
180 d = asdict(sample)
181 d["age_ms"] = age_ms
182 d["ok"] = bool(sample.ok and age_ms <= self.stale_ms)
183 if age_ms > self.stale_ms:
184 d["status"] = "STALE_RANGE"
185 return RangeSample(**d)
186
187 def as_json_line(self, peer: Optional[str] = None) -> str:
188 sample = self.get(peer)
189 if sample is None:
190 return json.dumps({"ok": False, "error": "NO_RANGE", "rate_source": "derived"})
191 return json.dumps(asdict(sample), separators=(",", ":"))
192
193 def as_text_line(self, peer: Optional[str] = None) -> str:
194 sample = self.get(peer)
195 if sample is None:
196 return "RANGE ok=0 error=NO_RANGE"
197 if not sample.ok:
198 return f"RANGE ok=0 error={sample.status} peer={sample.peer} age_ms={sample.age_ms}"
199 rate = "nan" if sample.range_rate_mps is None else f"{sample.range_rate_mps:.3f}"
200 return (
201 f"RANGE ok=1 peer={sample.peer} seq={sample.seq} "
202 f"range_cm={sample.range_cm} range_m={sample.range_m:.3f} "
203 f"rate_mps={rate} rate_valid={int(sample.range_rate_valid)} "
204 f"age_ms={sample.age_ms} status={sample.status}"
205 )

267 notif_handlers = {
268 (Gid.Ranging, OidRanging.Start): self._handle_range_ntf_handle_range_ntf,
269 }
270 self.client = Client(port=self.args.uwb_port, notif_handlers=notif_handlers)
271
272 print(f"UWB: initializing session {self.args.session_id} on {self.args.uwb_port}", flush=True)
273 rts, session_handle = self.client.session_init(self.args.session_id, SessionType.Ranging)
274 if rts != Status.Ok:

280 rts, msg = self.client.session_set_app_config(self.session_handle, configs)
281 if rts != Status.Ok:

286 if rts != Status.Ok:

292 if self.client is None:
293 return

312def arduino_serial_loop(args: argparse.Namespace, store: RangeStore, stop_event: threading.Event) -> None:
313 if args.no_arduino_port or not args.arduino_port:
314 print("Arduino serial API disabled; running UWB session only.", flush=True)
315 while not stop_event.is_set():
316 time.sleep(0.25)
317 return
318
319 print(f"Arduino API: opening {args.arduino_port} at {args.arduino_baud} baud", flush=True)
320 with serial.Serial(args.arduino_port, args.arduino_baud, timeout=0.2, write_timeout=1.0) as ser:
321 time.sleep(2.0) # Arduino boards often reset when the serial port opens.
322 ser.reset_input_buffer()
323 ser.write(b"READY UWB_GATEWAY\n")
324 while not stop_event.is_set():
325 raw = ser.readline()
326 if not raw:
327 continue
328 try:
329 line = raw.decode("utf-8", errors="replace").strip()
330 except Exception:
331 line = ""
332 if not line:
333 continue
334
335 parts = line.split()
336 cmd = parts[0].upper()
337 peer = parts[1] if len(parts) > 1 else None
338
339 if cmd in {"R", "GET", "GET_RANGE", "RANGE?"}:
340 response = store.as_json_line(peer) if args.json_to_arduino else store.as_text_line(peer)
341 elif cmd == "STATUS":
342 response = store.as_json_line(peer)
343 elif cmd == "HELP":
344 response = "CMDS GET_RANGE [peer], STATUS [peer], HELP"
345 else:
346 response = f"ERR unknown_command={cmd}"
347 ser.write((response + "\n").encode("utf-8"))

376def main() -> int:
377 args = build_arg_parser().parse_args()
378 load_qorvo_uci(args.sdk_tools)
379
380 store = RangeStore(stale_ms=args.stale_ms)
381 stop_event = threading.Event()
382
383 def request_stop(signum, frame): # noqa: ARG001
384 stop_event.set()
385
386 signal.signal(signal.SIGINT, request_stop)
387 signal.signal(signal.SIGTERM, request_stop)
388
389 session = UwbSession(args, store)
390 try:
391 session.start()
392 arduino_serial_loop(args, store, stop_event)
393 except KeyboardInterrupt:
394 stop_event.set()

396 print(f"ERROR: {exc}", file=sys.stderr, flush=True)
397 return 1
398 finally:
399 session.stop()
400 return 0
