34def load_bias(base, cols):
35 """Return (t_abs, bias Nx3) from a filter CSV, or None if unavailable.
36
37 Absolute (steady-clock) time is kept so the bias can be interpolated onto
38 the raw sensor timestamps, which share the same clock.
39 """
40 for path in (base, base + '.csv'):
41 if os.path.exists(path):
42 try:
43 d = pd.read_csv(path)
44 except Exception:
45 return None
46 if 'time' not in d.columns or not all(c in d.columns for c in cols):
47 return None
48 for c in ['time', *cols]:
49 d[c] = pd.to_numeric(d[c], errors='coerce')
50 d = d.dropna(subset=['time', *cols]).sort_values('time')
51 if len(d) < 2:
52 return None
53 return d['time'].to_numpy(float), d[cols].to_numpy(float)
54 return None

57def load_trans_accel(base):
58 """Return (t_abs, ax_raw Nx3, ba_body Nx3) from the TRANS-EKF log.
59
60 ax_raw (body-frame raw accel) and ba_body (accel bias already rotated into
61 the body frame in nav_node) live on the same rows, so unbiased = raw-ba_body
62 needs no interpolation. Returns None if the log/columns aren't present.
63 """
64 raw_c = ['ax_raw', 'ay_raw', 'az_raw']
65 bb_c = ['bax_b', 'bay_b', 'baz_b']
66 for path in (base, base + '.csv'):
67 if os.path.exists(path):
68 try:
69 d = pd.read_csv(path)
70 except Exception:
71 return None
72 need = ['time', *raw_c, *bb_c]
73 if not all(c in d.columns for c in need):
74 return None
75 for c in need:
76 d[c] = pd.to_numeric(d[c], errors='coerce')
77 d = d.dropna(subset=need).sort_values('time')
78 if len(d) < 2:
79 return None
80 return (d['time'].to_numpy(float),
81 d[raw_c].to_numpy(float),
82 d[bb_c].to_numpy(float))
83 return None

86def load_accel_q0(base):
87 """Return (t_abs, a_q0 Nx3, bias_q0 Nx3) from the TRANS-EKF log.
88
89 a_q0 = raw accel rotated into the q0 frame (done in nav_node), bias_q0 is
90 the accel bias state (already q0). unbiased = a_q0 - bias_q0 should sit at
91 the inertial gravity (grav_ref) at rest and stay put during pure rotation.
92 """
93 aq_c = ['ax_i', 'ay_i', 'az_i']
94 b_c = ['bax', 'bay', 'baz']
95 for path in (base, base + '.csv'):

99 except Exception:
100 return None
101 need = ['time', *aq_c, *b_c]
102 if not all(c in d.columns for c in need):
103 return None
104 for c in need:
105 d[c] = pd.to_numeric(d[c], errors='coerce')
106 d = d.dropna(subset=need).sort_values('time')
107 if len(d) < 2:
108 return None
109 return (d['time'].to_numpy(float),
110 d[aq_c].to_numpy(float),

115def overlay_bias(axes3, sub, bias_data, unbiased_label='unbiased'):
116 """Overlay bias estimate + unbiased (raw - bias) on 3 per-axis raw subplots.
117
118 axes3 : the three Axes (x, y, z) already showing the raw trace.
119 sub : the raw sensor dataframe (has absolute 'time' + v0/v1/v2).
120 bias_data : (t_abs, bias Nx3) from load_bias, or None (then no-op).
121 """
122 if bias_data is None or len(sub) == 0:
123 return
124 tb_abs, bias = bias_data
125 t_abs = sub['time'].to_numpy(float)
126 t0 = t_abs[0]
127 trel = t_abs - t0
128 for i in range(3):
129 raw_i = sub[f'v{i}'].to_numpy(float)
130 b_i = np.interp(t_abs, tb_abs, bias[:, i])
131 axes3[i].plot(trel, b_i, color='tab:red', linewidth=1.0, alpha=0.85,
132 linestyle='--', label='bias')
133 axes3[i].plot(trel, raw_i - b_i, color='black', linewidth=0.8,
134 alpha=0.7, label=unbiased_label)
135 axes3[i].legend(loc='upper right', fontsize=8)

138def extract(df, sensor_type):
139 sub = df[df['type'] == sensor_type].copy()
140 sub['time'] = pd.to_numeric(sub['time'], errors='coerce')
141 sub['v0'] = pd.to_numeric(sub['v0'], errors='coerce')
142 sub['v1'] = pd.to_numeric(sub['v1'], errors='coerce')
143 sub['v2'] = pd.to_numeric(sub['v2'], errors='coerce')
144 sub['v3'] = pd.to_numeric(sub['v3'], errors='coerce')
145 sub = sub.dropna(subset=['time']).sort_values('time').reset_index(drop=True)
146 return sub

149def rel_time(sub):
150 t = sub['time'].to_numpy(dtype=float, copy=True)
151 if len(t) > 0:
152 t -= t[0]

290 # ── Magnetometer calibration ellipses (raw ellipsoid -> calibrated sphere) ─
291 # Fits the ellipsoid on this log and shows raw vs calibrated 2-D projections.
292 # A good calibration turns the offset ellipse (raw) into a centred circle.
293 fig6 = None
294 if _HAS_CALIB and len(magnm) >= 10:
295 mx = magnm['v0'].to_numpy()
296 my = magnm['v1'].to_numpy()
297 mz = magnm['v2'].to_numpy()
298 try:
299 bias, W = fit_ellipsoid(mx, my, mz)
300 cx, cy, cz = apply_mag_calibration(mx, my, mz, bias, W)
301 raw = np.column_stack([mx, my, mz])
302 raw_norm = np.linalg.norm(raw, axis=1)
303 cal_norm = np.sqrt(cx**2 + cy**2 + cz**2)
304 r_raw = np.linalg.norm(raw - bias, axis=1).mean() # fitted sphere radius
305 r_cal = cal_norm.mean()

307 def _circle(ax, cen, r, color):
308 th = np.linspace(0, 2 * np.pi, 200)
309 ax.plot(cen[0] + r * np.cos(th), cen[1] + r * np.sin(th),
310 color=color, linewidth=1.2, linestyle='--')
311
312 fig6, axes6 = plt.subplots(2, 3, figsize=(15, 9))
313 fig6.suptitle('Magnetometer Calibration — Raw ellipsoid vs '
314 'Calibrated sphere (2-D projections)', fontsize=14)
315 proj = [('XY', 0, 1), ('XZ', 0, 2), ('YZ', 1, 2)]
316 raw_d = [mx, my, mz]
317 cal_d = [cx, cy, cz]
318 labels = ['mx', 'my', 'mz']
319 for k, (name, i, j) in enumerate(proj):
320 a = axes6[0, k]
321 a.scatter(raw_d[i], raw_d[j], s=2, alpha=0.35, color='tab:blue')
322 a.plot(bias[i], bias[j], 'kx', markersize=9,
323 label='hard-iron center')
324 _circle(a, (bias[i], bias[j]), r_raw, 'tab:red')
325 a.set_title(f'Raw — {name}')
326 a.set_xlabel(labels[i])
327 a.set_ylabel(labels[j])
328 a.set_aspect('equal', adjustable='datalim')
329 a.grid(True, linestyle='--', alpha=0.4)
330 if k == 0:
331 a.legend(loc='upper right', fontsize=8)

344 def _pp(x):
345 return 100 * (x.max() - x.min()) / 2 / x.mean()
