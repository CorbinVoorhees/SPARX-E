23def main(path, nis_dim):
24 df = pd.read_csv(path)

27 required = [

34 # 'time' is absolute steady_clock seconds -> elapsed from first row

37 # Figure 1: gyro, rows = x/y/z, cols = raw / corrected / bias / mean
38 fig, ax = plt.subplots(3, 4, figsize=(18, 10), sharex=True)

41 plot_one(ax[i, 1], t, df, f"gunb{c}", f"gyro corrected {c}", "rad/s")
42 plot_one(ax[i, 2], t, df, f"bg{c}", f"gyro bias {c}", "rad/s")

50 # Figure 2: attitude (radians, as logged)
