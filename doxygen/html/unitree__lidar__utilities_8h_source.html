96inline uint32_t crc32(const uint8_t *buf, uint32_t len)
97{
98 uint8_t i;
99 uint32_t crc = 0xFFFFFFFF;
100 while (len--)
101 {
102 crc ^= *buf++;
103 for (i = 0; i < 8; ++i)
104 {
105 if (crc & 1)
106 crc = (crc >> 1) ^ 0xEDB88320;
107 else
108 crc = (crc >> 1);
109 }
110 }
111 return ~crc;
112}

122inline void parseFromPacketToPointCloud(
123 PointCloudUnitree &cloud,
124 const LidarPointDataPacket &packet,
125 bool use_system_timestamp = true,
126 float range_min = 0,
127 float range_max = 100
128 )
129{
130 // scan info
131 const int num_of_points = packet.data.point_num;
132 const float time_step = packet.data.time_increment;
133 const float scan_period = packet.data.scan_period;
134
135 // intermediate variables
136 const float sin_beta = sin(packet.data.param.beta_angle);
137 const float cos_beta = cos(packet.data.param.beta_angle);
138 const float sin_xi = sin(packet.data.param.xi_angle);
139 const float cos_xi = cos(packet.data.param.xi_angle);
140 const float cos_beta_sin_xi = cos_beta * sin_xi;
141 const float sin_beta_cos_xi = sin_beta * cos_xi;
142 const float sin_beta_sin_xi = sin_beta * sin_xi;
143 const float cos_beta_cos_xi = cos_beta * cos_xi;
144
145 // cloud init
146 if (use_system_timestamp)
147 {
148 cloud.stamp = getSystemTimeStamp() - scan_period;
149 }else{
150 cloud.stamp = packet.data.info.stamp.sec + packet.data.info.stamp.nsec / 1.0e9;
151 }
152 cloud.id = 1;
153 cloud.ringNum = 1;
154 cloud.points.clear();
155 cloud.points.reserve(300);
156
157 // transform raw data to a pointcloud
158 auto &ranges = packet.data.ranges;
159 auto &intensities = packet.data.intensities;
160
161 float time_relative = 0;
162 float alpha_cur = packet.data.angle_min + packet.data.param.alpha_angle_bias;
163 float alpha_step = packet.data.angle_increment;
164 float theta_cur = packet.data.com_horizontal_angle_start + packet.data.param.theta_angle_bias;
165 float theta_step = packet.data.com_horizontal_angle_step;
166
167 float range_float;
168 float sin_alpha, cos_alpha, sin_theta, cos_theta;
169 float A, B, C;
170
171 PointUnitree point3d;
172 point3d.ring = 1;
173 // std::cout << "packet.data.param.range_scale = " << packet.data.param.range_scale << std::endl;
174
175 for (int j = 0; j < num_of_points; j += 1, alpha_cur += alpha_step,
176 theta_cur += theta_step, time_relative += time_step)
177 {
178 // jump invalid points
179 if (ranges[j] < 1)
180 {
181 continue;
182 }
183
184 // calculate point range in float type
185 range_float = packet.data.param.range_scale * ((float)ranges[j] + packet.data.param.range_bias);
186
187 // jump points beyond range limit
188 if ( range_float < packet.data.range_min || range_float > packet.data.range_max)
189 {
190 continue;
191 }
192
193 // jump points beyond range limit
194 if (range_float < range_min || range_float > range_max)
195 {
196 continue;
197 }
198
199 // transform to XYZ coordinate
200 sin_alpha = sin(alpha_cur);
201 cos_alpha = cos(alpha_cur);
202 sin_theta = sin(theta_cur);
203 cos_theta = cos(theta_cur);
204
205 A = (-cos_beta_sin_xi + sin_beta_cos_xi * sin_alpha) * range_float + packet.data.param.b_axis_dist;
206 B = cos_alpha * cos_xi * range_float;
207 C = (sin_beta_sin_xi + cos_beta_cos_xi * sin_alpha) * range_float;
208
209 point3d.x = cos_theta * A - sin_theta * B;
210 point3d.y = sin_theta * A + cos_theta * B;
211 point3d.z = C + packet.data.param.a_axis_dist;
212
213 // push back this point to cloud
214 point3d.intensity = intensities[j];
215 point3d.time = time_relative;
216 cloud.points.push_back(point3d);
217 }
218}

292 point3d.x = 0;

295

298 point3d.time = time_relative;
