312 Eigen::VectorXd imu_pseudo = latest_imu; // 3-vector: [ax, ay, az]
313

316 q_anchor_meas.normalize();

320 rotational_mekf(q_anchor_meas, latest_gyro,
321 prior_orientation, prior_P_rot,
322 dt, sigma_gyro,
323 new_P_rot, new_orientation);
324
325 if (new_orientation.allFinite())

327 prior_orientation = new_orientation;
328 prior_P_rot = new_P_rot;
329 }
330 else
331 exit(2);
332
333 Eigen::Quaterniond quat(prior_orientation[0], prior_orientation[1],
334 prior_orientation[2], prior_orientation[3]);
335 quat.normalize();
336 auto R = quat.toRotationMatrix();
337 double roll = std::atan2(R(2, 1), R(2, 2));
338 double s = -R(2, 0);
339 double pitch = (std::abs(s) >= 1.0) ? std::copysign(MY_PI / 2.0, s) : std::asin(s);
340 double yaw = std::atan2(R(1, 0), R(0, 0));
341 Eigen::Vector3d orientation(roll, pitch, yaw);
342
343 Eigen::VectorXd control(4);

353 prior_location, prior_P_loc, prior_P_rot, dt,

356 // std::cout << new_location << std::endl;

360 prior_P_loc = new_P_loc;

365 }
366
367 // Update the reference time so the next timer tick computes a correct dt
368 last_uwb = now;
369
370 RCLCPP_INFO_THROTTLE(this->get_logger(), *this->get_clock(), 500,
371 "EKF OUT | pos=[%.3f, %.3f, %.3f], ranging=[%.3f, %.3f], a_inertial=[%.3f, %.3f, %.3f], time=%.3f",
372 prior_location[0], prior_location[1], prior_location[2],
373 range - prior_location[9], range_rate - prior_location[10],
374 latest_imu[0] - prior_location[6],
375 latest_imu[1] - prior_location[7],
376 latest_imu[2] - prior_location[8],
377 std::chrono::duration<double>(now.time_since_epoch()).count());

379 publish_estimate_locked(prior_P_loc, prior_P_rot);
