41 void predict(double dt) override {
42 if (!mekf_ref->is_finished_orienting())
43 return;
44
45 const double wr = control_snapshot.z(0);
46 const double wl = control_snapshot.z(1);
47
48 const Eigen::Matrix3d R_I_B =
49 mekf_ref->rel_orientation().normalized().toRotationMatrix();
50
51 position() += velocity() * dt;
52 velocity() =
53 R_I_B * V3(GEOM.rw / 2 * (mu_r() * wr + mu_l() * wl), 0.0, 0.0);
54
55 Eigen::Matrix<double, 8, 8> Phi = Eigen::Matrix<double, 8, 8>::Zero();
56 Phi.block<3, 3>(0, 0) = Eigen::Matrix3d::Identity();
57 Phi.block<3, 3>(0, 3) = Eigen::Matrix3d::Identity() * dt;
58 Phi.block<3, 1>(3, 6) = R_I_B.col(0) * GEOM.rw / 2 * wr;
59 Phi.block<3, 1>(3, 7) = R_I_B.col(0) * GEOM.rw / 2 * wl;
60 Phi(6, 6) = 1.0;
61 Phi(7, 7) = 1.0;
62
63 Eigen::MatrixXd Qd = Eigen::MatrixXd::Zero(8, 8);
64 Qd.diagonal() << 1e-6, 1e-6, 1e-6, 1e-4, 1e-3, 1e-4, 1e-4, 1e-4;
65 Qd *= dt;
66
67 P = Phi * P * Phi.transpose() + Qd;
68 }
