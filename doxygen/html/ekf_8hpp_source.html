36 // generic Kalman correction
37 void correct(const Eigen::MatrixXd &H, const Eigen::MatrixXd &R,
38 const Eigen::VectorXd &d_r, std::string name = "") {
39 const Eigen::MatrixXd Pz = H * P * H.transpose();
40 const Eigen::MatrixXd S = Pz + R;
41 const Eigen::MatrixXd K = P * H.transpose() * S.inverse();
42 const Eigen::VectorXd dx = K * d_r;
43 nis = (d_r.transpose() * S.inverse() * d_r)(0, 0);
44
45 // stashed pre-gate so rejected updates still show their innovation
46 last_r_[name] = d_r;
47
48 // do a one-tailed chi-score test
49 double chi_sq_threshold = 100.0;
50
51 // lowered the ci from 95 -> 75
52 if (d_r.size() == 1)
53 chi_sq_threshold = 1.323; // 75% confidence
54 else if (d_r.size() == 2)
55 chi_sq_threshold = 2.773;
56 else if (d_r.size() == 3)
57 chi_sq_threshold = 4.108;
58
59 // yeah no way this happened.
60 if (nis > chi_sq_threshold)
61 return;
62
63 // per-state gain magnitude of the applied correction (row-wise norm of K)
64 last_K_[name] = K.rowwise().norm();
65
66 print_diagnostic(S, Pz, R, d_r, name);
67 const Eigen::MatrixXd I = Eigen::MatrixXd::Identity(P.rows(), P.cols());
68 const Eigen::MatrixXd IKH = I - K * H;
69 P = IKH * P * IKH.transpose() + K * R * K.transpose();
70 apply_error(dx);
71 ++steps;
72 }
