37 def _get(self, d: dict, path: str, default: Any = np.nan) -> Any:
38 cur: Any = d
39 for p in path.split("."):
40 if not isinstance(cur, dict) or p not in cur or cur[p] is None:
41 return default
42 cur = cur[p]
43 return cur

45 def _scalar_series(self, blocks, path: str):
46 vals = []
47 for b in blocks:
48 v = self._get(b, path, np.nan)
49 if isinstance(v, (int, float, np.integer, np.floating)):
50 vals.append(float(v))
51 else:
52 vals.append(np.nan)
53 return np.asarray(vals, dtype=float)

55 def _time(self, blocks):
56 t = self._scalar_series(blocks, "time.t")
57 if len(t) == 0 or np.all(np.isnan(t)):
58 return t
59 if np.nanmin(t) > 1e5:
60 t = t - t[0]
61 return t

78 def _plot_series(self, ax, t, y, label="value", color=None, linewidth=None):
79 mask = np.isfinite(t) & np.isfinite(y)
80 if np.any(mask):
81 ax.plot(
82 t[mask],

91 def _plot_with_cov(self, ax, t, y, cov=None, label="value", color=None):
92 mask = self._plot_series(ax, t, y, label=label, color=color)
93
94 if cov is not None and len(cov) == len(y):
95 cov_mask = mask & np.isfinite(cov)
96 if np.any(cov_mask):
97 sigma = np.sqrt(np.maximum(cov[cov_mask], 0.0))
98 ax.fill_between(
99 t[cov_mask],
100 y[cov_mask] - 2.0 * sigma,
101 y[cov_mask] + 2.0 * sigma,
102 alpha=self.alpha,
103 color=color,
104 label=r"$\pm 2\sigma$",
105 zorder=1,
106 )
107
108 self._format_ax(ax)

110 def _plot_mean_std(self, ax, t, mean, std, label="Mean", color=None):
111 mean_mask = np.isfinite(t) & np.isfinite(mean)
112
113 if std is not None and len(std) == len(mean):
114 std_mask = mean_mask & np.isfinite(std)
115 if np.any(std_mask):
116 ax.fill_between(
117 t[std_mask],
118 mean[std_mask] - 2.0 * std[std_mask],
119 mean[std_mask] + 2.0 * std[std_mask],
120 alpha=self.alpha,
121 color=color,
122 label=r"$\pm 2\sigma$",
123 zorder=1,
124 )
125
126 if np.any(mean_mask):
127 ax.plot(
128 t[mean_mask],
129 mean[mean_mask],
130 label=label,
131 color=color,
132 linewidth=3.5,
133 zorder=5,
134 )
135
136 self._format_ax(ax)
