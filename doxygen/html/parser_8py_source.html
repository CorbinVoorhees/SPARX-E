24 def __extract_blocks(self, text, title: str):
25 """
26 Returns all blocks matching:
27
28 ========== TITLE ==========
29 ...
30 =====================================
31 """
32 title_b = self.__to_bytes(title)
33
34 pattern = (
35 rb"=+\s*" + re.escape(title_b) + rb"\s*=+"
36 rb"(.*?)"
37 rb"(?=\n=+\s*[A-Z][A-Z _]*\s*=+|\Z)"
38 )
39
40 return [m.group(1) for m in re.finditer(pattern, text, flags=re.S)]

42 def __extract_section(self, block, section_name: str):
43 """
44 Extracts:

52 pattern = (

61 def __parse_keyvals(self, text, label: str):
62 """
63 Parses:
64
65 pos : x=1.0, y=2.0, z=3.0
66 rangeDot : raw=0.1, mean=0.2, std=0.3
67 time : t=123.0, dt=0.025
68 """
69 label_b = self.__to_bytes(label)
70
71 pattern = re.escape(label_b) + rb"\s*:\s*(.+)"
72 match = re.search(pattern, text)
73
74 if not match:
75 return None
76
77 line = match.group(1)
78
79 pairs = re.findall(
80 rb"([A-Za-z_]\w*)\s*=\s*(" + _FLOAT_RE + rb")",
81 line
82 )
83
84 return {k.decode("utf-8"): float(v) for k, v in pairs}

86 def __parse_matrix(self, text, label: str):
87 """
88 Parses:
89
90 K:
91 1 2 3
92 4 5 6
93
94 Stops at next matrix-ish label, section, block, or EOF.
95 """
96 label_b = self.__to_bytes(label)
97
98 pattern = (
99 re.escape(label_b)
100 + rb"\s*:\s*\n"
101 + rb"(.*?)"
102 + rb"(?=\n[A-Za-z_]\w*\s*:|\n---|\n=+|\Z)"
103 )
104
105 match = re.search(pattern, text, flags=re.S)
106
107 if not match:
108 return None
109
110 rows = []
111
112 for line in match.group(1).splitlines():
113 nums = re.findall(_FLOAT_RE, line)
114 if nums:
115 rows.append([float(x) for x in nums])
116
117 if not rows:
118 return None
119
120 # --- NEW CODE: Enforce homogeneous shape ---
121 # Assume the length of the first parsed row is the correct matrix width
122 expected_len = len(rows[0])
123
124 # Filter out any malformed rows (like stray log lines)
125 valid_rows = [r for r in rows if len(r) == expected_len]
126
127 if not valid_rows:
128 return None
129
130 return np.array(valid_rows, dtype=float)

309 def parse_logs(self):
310 out = {}
311
312 ekf_text = self.__read_mmap(self.ekf_log_path)
313 trans_blocks = self.__extract_blocks(ekf_text, "TRANSLATIONAL EKF")
314 out["translational"] = [self.__parse_trans_block(b) for b in trans_blocks]
315
316 nav_text = self.__read_mmap(self.kalman_state_log)
317 nav_blocks = self.__extract_blocks(nav_text, "NAV NODE OUT")
318 out["nav"] = [self.__parse_nav_block(b) for b in nav_blocks]
319
320 if self.mekf_log_path is not None:
321 mekf_text = self.__read_mmap(self.mekf_log_path)
322 mekf_blocks = self.__extract_blocks(mekf_text, "MEKF OUT")
323 out["mekf"] = [self.__parse_mekf_block(b) for b in mekf_blocks]
324
325 return out
