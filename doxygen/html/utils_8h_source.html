23std::string string_format(const char *fmt, Args... args) {
24 // obtain theoretical exact container size required for all variable
25 // substitutions
26 int size = std::snprintf(nullptr, 0, fmt, args...);
27 if (size <= 0)
28 return "";
29

31 std::string result(size, '\0');

122// Reads `expected_count` whitespace/newline-separated doubles from `path`.
123// Returns false (leaving `out` untouched) if the file is missing or short.
124inline bool load_doubles(const std::string &path, std::vector<double> &out,
125 size_t expected_count) {
126 std::ifstream in(path);
127 if (!in.is_open())
128 return false;
129
130 std::vector<double> vals;
131 vals.reserve(expected_count);
132 double v;
133 while (vals.size() < expected_count && (in >> v))
134 vals.push_back(v);
135
136 if (vals.size() != expected_count)
137 return false;
138
139 out = std::move(vals);
140 return true;
141}

184 // crosses T -> double, doesn't fit the macro's "both branches return T"
185 // shape, so this is a plain if constexpr rather than IF_CONSTEXPR_ASSIGN
186 static double deviation_like(const T &x, const T &med) {
187 if constexpr (std::is_arithmetic_v<T>)
188 return std::abs(x - med);
189 else
190 return (x - med).norm();

193 static T sign_like(const T &x, const T &med) {
194 T dummy;
195 IF_CONSTEXPR_ASSIGN(dummy, T,
196 (x - med) > 0 ? 1.0 : ((x - med) < 0 ? -1.0 : 0.0),
197 (x - med).cwiseSign());

201 // true = reject. med_/mad_ are updated only on accepted samples — a
202 // sustained spike train must never drag the reference toward itself
203 // (otherwise the detector "masks" and stops rejecting).
204 bool check_spike(const T &x) {
205 if (!this->spike_primed_) {
206 this->med_ = x;
207 this->mad_ = 0.0;
208 this->spike_primed_ = true;
209 return false;
210 }
211
212 const double dev = deviation_like(x, this->med_);
213 // mad_ > 1e-12 guard: while mad_ is still ~0 (first couple of accepted
214 // samples) the raw test `dev > k*mad_` would reject almost everything
215 // that isn't bit-identical to med_ — always accept until mad_ settles.
216 const bool reject =
217 (this->mad_ > 1e-12) && (dev > this->spike_cfg_.k * this->mad_);
218 if (!reject) {
219 this->med_ += this->spike_cfg_.eta * sign_like(x, this->med_);
220 const double dev2 = deviation_like(x, this->med_);
221 this->mad_ += this->spike_cfg_.eta * (dev2 - this->mad_);
222 }
223 return reject;
224 }

262
263 if (this->count == 1) {
264 this->mean = x;
265 return;
266 }
267
268 const double n = static_cast<double>(this->count);
269 const T delta = x - this->mean;
270
271 this->mean += delta / n;
272
273 const T delta2 = x - this->mean;
274
275 this->var = ((n - 2.0) / (n - 1.0)) * this->var +
276 product_like(delta, delta2) / (n - 1.0);
277 }
