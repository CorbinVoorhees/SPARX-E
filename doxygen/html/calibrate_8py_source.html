36def fit_ellipsoid(mx, my, mz):
37 """
38 Fit an ellipsoid to 3D magnetometer data via least squares.
39 Returns (bias, W) where bias is the hard-iron offset vector and W
40 is the 3×3 soft-iron correction matrix.
41
42 Apply with: m_cal = W @ (m_raw - bias)
43 """
44 D = np.column_stack([
45 mx**2, my**2, mz**2,
46 2*mx*my, 2*mx*mz, 2*my*mz,
47 2*mx, 2*my, 2*mz
48 ])
49 ones = np.ones(len(mx))
50 v, _, _, _ = np.linalg.lstsq(D, ones, rcond=None)
51
52 a, b, c, d, e, f, g, h, k = v
53 M = np.array([[a, d, e],
54 [d, b, f],
55 [e, f, c]])
56 n = np.array([g, h, k])
57
58 # Hard-iron bias: center of the fitted ellipsoid
59 bias = -np.linalg.solve(M, n)
60
61 # Normalize so that the centered ellipsoid equation equals 1
62 scale = 1.0 + n @ np.linalg.solve(M, n)
63 A_norm = M / scale
64
65 # Soft-iron W maps the ellipsoid to a unit sphere via W = A_norm^(1/2)
66 eigvals, eigvecs = np.linalg.eigh(A_norm)
67 eigvals = np.maximum(eigvals, 0.0) # guard against tiny negative numerics
68 W = eigvecs @ np.diag(np.sqrt(eigvals)) @ eigvecs.T
69
70 # W above maps calibrated output onto a *unit* sphere — rescale so it
71 # lands back in the raw data's physical units (Tesla) instead.
72 raw = np.column_stack([mx, my, mz])
73 r0 = np.linalg.norm(raw - bias, axis=1).mean()
74 W = r0 * W
75
76 return bias, W

79def apply_mag_calibration(mx, my, mz, bias, W):
80 raw = np.column_stack([mx, my, mz]) - bias # hard-iron correction
81 cal = (W @ raw.T).T # soft-iron correction
82 return cal[:, 0], cal[:, 1], cal[:, 2]

85def save_mag_calibration(path, bias, W):
86 """
87 Flat text format consumed by MEKF::load_mag_calibration (mekf.hpp):

96def recommend_cutoff(x, fs, top_frac=0.2, drop_factor=3.0, fmin_floor=0.5):
97 """
98 Welch PSD -> noise floor = median PSD over the top `top_frac` of the
99 band -> cutoff = first frequency from DC where the PSD has decayed to
100 within `drop_factor` of that floor.
101 """
102 nperseg = min(len(x), 2048)
103 f, Pxx = welch(x, fs=fs, nperseg=nperseg)
104 n_top = max(1, int(len(f) * top_frac))
105 noise_floor = np.median(Pxx[-n_top:])
106 threshold = noise_floor * drop_factor
107 below = np.nonzero(Pxx <= threshold)[0]
108 fc = f[below[0]] if len(below) > 0 else f[-1]
109 return max(fc, fmin_floor), f, Pxx, noise_floor

112def recommend_sensor_cutoff(df, sensor_type, fs):
113 """Per-axis recommend_cutoff; reports the max across the 3 axes
114 (conservative — a too-low cutoff clips real bandwidth on the noisiest
115 axis, which is worse than slightly under-filtering the quieter ones)."""
116 sub = extract(df, sensor_type)
117 if len(sub) < 16:
118 return None, None
119
120 psd_data = []
121 for i in range(3):
122 x = sub[f'v{i}'].to_numpy()
123 fc, f, Pxx, floor = recommend_cutoff(x, fs)
124 psd_data.append((f, Pxx, floor, fc))
125
126 fc_sensor = max(d[3] for d in psd_data)
127 return fc_sensor, psd_data

135def plot_psd(sensor_type, psd_data):
136 fig, axes = plt.subplots(3, 1, figsize=(12, 9), sharex=True)

152def main(csv_path=SENSORS_CSV, show_plots=True):
153 df = load_csv(csv_path)
154 if 'type' not in df.columns:
155 raise KeyError("CSV missing 'type' column — is this a sensors raw log?")
156
157 # ── Magnetometer calibration ─────────────────────────────────────────────
158 magnm = extract(df, 'magnm')
159 if len(magnm) >= 10:
160 mx = magnm['v0'].to_numpy()
161 my = magnm['v1'].to_numpy()
162 mz = magnm['v2'].to_numpy()
163 try:
164 bias, W = fit_ellipsoid(mx, my, mz)
165 print(f"Hard-iron bias : {bias}")
166 print(f"Soft-iron W :\n{W}")
167 save_mag_calibration(MAG_CALIBRATION_PATH, bias, W)
168 print(f"Saved magnetometer calibration to {MAG_CALIBRATION_PATH}")
169 except np.linalg.LinAlgError as exc:
170 print(f"Ellipsoid fit failed: {exc}")
171 else:
172 print("Not enough magnetometer samples for calibration (need >= 10)")
173
174 # ── Filter cutoff recommendation ─────────────────────────────────────────
175 fc_gyro, psd_gyro = recommend_sensor_cutoff(df, 'gyro', IMU_SAMPLE_RATE_HZ)
176 fc_accel, psd_accel = recommend_sensor_cutoff(df, 'accel', IMU_SAMPLE_RATE_HZ)
177 fc_magnm, psd_magnm = recommend_sensor_cutoff(df, 'magnm', IMU_SAMPLE_RATE_HZ)
178
179 if None not in (fc_gyro, fc_accel, fc_magnm):
180 save_filter_cutoffs(FILTER_CUTOFFS_PATH, fc_gyro, fc_accel, fc_magnm)
181 print(f"Recommended cutoffs (Hz): gyro={fc_gyro:.2f} "
182 f"accel={fc_accel:.2f} magnm={fc_magnm:.2f}")
183 print(f"Saved filter cutoffs to {FILTER_CUTOFFS_PATH}")
184 else:
185 print("Not enough samples to recommend filter cutoffs for all sensors")
186
187 if show_plots:
188 if psd_gyro:
189 plot_psd('gyro', psd_gyro)
190 if psd_accel:
191 plot_psd('accel', psd_accel)
192 if psd_magnm:
193 plot_psd('magnm', psd_magnm)
194 plt.show()
