43 // Checks if the joystick has published a value in the last 0.5 seconds, if

52 // Command loop pulls values from either manual or autonomy commands then
53 // sends values to arduino Loop runs at 50Hz
54 void command_loop() {
55 if (manual_operation) {
56 double x = linear_x;
57 double z = angular_z;
58 if (abs(x) < 0.05 && abs(z) < 0.15) {
59 rover_speed_manual = 0.0;
60 rover_rotation_speed_manual = 0.0;
61 } else if (abs(x) < 0.05 && abs(z) >= 0.15) {
62 rover_speed_manual = 0.0;
63 rover_rotation_speed_manual = z;
64 } else if (abs(x) >= 0.05 && abs(z) < 0.15) {
65 rover_speed_manual = x;
66 rover_rotation_speed_manual = 0.0;
67 } else {
68 rover_speed_manual = x;
69 rover_rotation_speed_manual = z;
70 }
71 send_command_to_arduino("MOVE:" + std::to_string(rover_speed_manual) +
72 "," +
73 std::to_string(rover_rotation_speed_manual));
74 } else {
75 std::cout << "AUTO:" + std::to_string(this->w_r) + "," +
76 std::to_string(this->w_l)
77 << std::endl;
78 send_command_to_arduino("AUTO:" + std::to_string(this->w_r) + "," +
79 std::to_string(this->w_l));
80 }
