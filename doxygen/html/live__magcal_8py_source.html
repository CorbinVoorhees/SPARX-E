1"""Live magnetometer calibration-coverage viewer.
2
3Run nav_filters (writing sparxe_sensors_raw.csv) in one terminal, this in
4another, then pick the robot up and tumble it through all orientations. The
5three projections fill in live; the title turns green when the point cloud
6covers the sphere well enough to calibrate. Then run calibrate.py on the log.
7
8 python live_magcal.py [path_to_sparxe_sensors_raw.csv]
9
10Press 'q' to quit.
11"""

18import numpy as np
19import pandas as pd
20import matplotlib.pyplot as plt
21from matplotlib.animation import FuncAnimation
22
23CSV = sys.argv[1] if len(sys.argv) > 1 else '../sparxe_sensors_raw.csv'
24INTERVAL_MS = 500
25PROJ = [('XY', 0, 1), ('XZ', 0, 2), ('YZ', 1, 2)]
26LABELS = ['mx', 'my', 'mz']
27
28# coverage thresholds for the GO verdict
29GOOD_FLATNESS = 0.60 # smallest/largest cloud extent (sphere ~1, arc ~0)
30GOOD_AXISRATIO = 0.50 # min/max per-axis span
31
32
33def read_mag(path):
34 if not os.path.exists(path):
35 return None
36 try:
37 df = pd.read_csv(path, on_bad_lines='skip')
38 except Exception:
39 return None
40 if 'type' not in df.columns or not {'v0', 'v1', 'v2'}.issubset(df.columns):
41 return None
42 m = df[df['type'] == 'magnm']
43 if len(m) == 0:
44 return None
45 M = m[['v0', 'v1', 'v2']].apply(pd.to_numeric, errors='coerce').dropna().to_numpy()
46 return M if len(M) else None
47
48
49def coverage(M):
50 rng = M.max(0) - M.min(0)
51 ratio = rng.min() / rng.max() if rng.max() > 0 else 0.0
52 sv = np.linalg.svd(M - M.mean(0), compute_uv=False)
53 flat = sv[2] / sv[0] if sv[0] > 0 else 0.0
54 return ratio, flat
55
56
57def verdict_text(ratio, flat):
58 good = flat > GOOD_FLATNESS and ratio > GOOD_AXISRATIO
59 msg = ('GOOD sphere — ready to calibrate' if good
60 else 'keep tumbling — cover ALL axes (rotate about x, y, z)')
61 return good, msg

64def main(csv=CSV):
65 fig, axes = plt.subplots(1, 3, figsize=(16, 5.5))
66 scatters = []
67 for ax, (name, i, j) in zip(axes, PROJ):
68 scatters.append(ax.scatter([], [], s=3, alpha=0.35, color='tab:blue'))
69 ax.set_title(name)
70 ax.set_xlabel(LABELS[i])
71 ax.set_ylabel(LABELS[j])
72 ax.set_aspect('equal', adjustable='box')
73 ax.grid(True, linestyle='--', alpha=0.4)
74
75 def update(_frame):
76 M = read_mag(csv)
77 if M is None:
78 fig.suptitle(f'Live Mag Calibration — waiting for magnm data in '
79 f'{csv} …', color='tab:orange', fontsize=13)
80 return scatters
81 for sc, (name, i, j) in zip(scatters, PROJ):
82 sc.set_offsets(np.column_stack([M[:, i], M[:, j]]))
83 for ax, (name, i, j) in zip(axes, PROJ):
84 for setlim, k in ((ax.set_xlim, i), (ax.set_ylim, j)):
85 lo, hi = M[:, k].min(), M[:, k].max()
86 pad = 0.1 * (hi - lo) if hi > lo else 1e-6
87 setlim(lo - pad, hi + pad)
88 ratio, flat = coverage(M)
89 good, msg = verdict_text(ratio, flat)
90 fig.suptitle(f'Live Mag Calibration — {len(M)} pts | flatness={flat:.2f} '
91 f'(>{GOOD_FLATNESS}) axis-ratio={ratio:.2f} '
92 f'(>{GOOD_AXISRATIO}) | {msg}',
93 color=('tab:green' if good else 'tab:red'), fontsize=13)
94 return scatters

96 def on_key(event):
97 if event.key in ('q', 'Q', 'escape'):
98 plt.close('all')

100 fig.canvas.mpl_connect('key_press_event', on_key)
101 # keep a ref so the animation isn't garbage-collected
102 fig._ani = FuncAnimation(fig, update, interval=INTERVAL_MS, blit=False,
103 cache_frame_data=False)
104 plt.tight_layout()
