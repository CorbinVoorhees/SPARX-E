114 void compute_shortest_path()
115 {
116 while (this->U.size() > 0 &&
117 (this->U[0].first < this->calculate_key(this->start) ||
118 !float_eq((*this->grid)[this->start].rhs, (*this->grid)[this->start].g)))
119 {
120 auto top = this->U.pop();
121 coordf k_old = top.first;
122 coordi u = top.second;
123
124 coordf k_new = this->calculate_key(u);
125
126 if (k_old < k_new)
127 {
128 // key was stale; reinsert with updated key
129 this->U.push({k_new, u});
130 continue;
131 }
132
133 bool overconsistent = ((*this->grid)[u].g > (*this->grid)[u].rhs);
134
135 if (overconsistent)
136 {
137 // g := rhs, then update predecessors only
138 (*this->grid)[u].g = (*this->grid)[u].rhs;
139
140 std::array<coordi, 9> ns;
141 uint32_t count = 0;
142 this->neighbors(u, ns, count, /*include_s=*/false);
143 for (uint32_t i = 0; i < count; i++)
144 this->update_vertex(ns[i]);
145 }
146 else
147 {
148 // underconsistent: g := infty, update predecessors AND u itself
149 (*this->grid)[u].g = INFTY;
150
151 std::array<coordi, 9> ns;
152 uint32_t count = 0;
153 this->neighbors(u, ns, count, /*include_s=*/true);
154 for (uint32_t i = 0; i < count; i++)

160 void update_vertex(coordi u)
161 {
162 if (u != this->goal)
163 {
164 std::array<coordi, 9> ns;

208 float cost(coordi a, coordi b)
209 {
210 if (this->grid->is_obstacle(a) || this->grid->is_obstacle(b))
211 return INFTY;
212
213 // Octile step distance: 1 for cardinal, sqrt(2) for diagonal
214 auto diff = b - a;
215 bool diagonal = (std::abs(diff.x) == 1) && (std::abs(diff.y) == 1);
216 float dist_cost = diagonal ? 1.41421356f : 1.0f;
217
218 // Spline-following bias: cells far from the spline cost more
219 float spline_penalty = (*this->grid)[b].dist_to_spline;
220
221 // Solar / illumination penalty
222 float solar_penalty = ((*this->grid)[b].illumnation < 0.5f) ? 0.5f : 0.0f;
223
224 // Elevation: only penalize uphill (downhill is free; no negative weights)
225 float dz = (*this->grid)[b].elevation - (*this->grid)[a].elevation;
226 float elevation_penalty = (dz > 0.0f) ? dz : 0.0f;
227
228 return dist_cost + spline_penalty + solar_penalty + elevation_penalty;
229 }

231 float heuristic(coordi a, coordi b)
232 {
233 // Octile heuristic — admissible for 8-connected uniform cost.
234 // Manhattan would over-estimate diagonal moves and break consistency.
235 float dx = static_cast<float>(std::abs(a.x - b.x));
236 float dy = static_cast<float>(std::abs(a.y - b.y));
237 float dmin = std::min(dx, dy);
238 float dmax = std::max(dx, dy);
239 return (1.41421356f - 1.0f) * dmin + dmax;
240 }

242 // BFS reachability check. Resets has_visited on the way out so the grid
243 // is clean for the actual planner.
244 bool path_exists(coordi src, coordi dst)
245 {
246 auto [rows_sz, cols_sz] = this->grid->size();
247 std::deque<coordi> q;
248 std::vector<coordi> touched;
249 q.push_back(src);
250 (*this->grid)[src].has_visited = true;
251 touched.push_back(src);
252
253 bool found = false;
254 while (!q.empty())
255 {
256 coordi curr = q.front();
257 q.pop_front();
258 if (curr == dst) { found = true; break; }
259
260 for (coordi d : directions)
261 {
262 coordi next = curr + d;
263 if (next.x < 0 || next.x >= (int32_t)cols_sz ||
264 next.y < 0 || next.y >= (int32_t)rows_sz)
265 continue;
266 if ((*this->grid)[next].has_visited || this->grid->is_obstacle(next))
267 continue;
268
269 (*this->grid)[next].has_visited = true;
270 touched.push_back(next);
271 q.push_back(next);
272 }
273 }
274
275 // restore has_visited so we don't poison subsequent searches
276 for (const coordi &c : touched)
277 (*this->grid)[c].has_visited = false;
278
279 return found;
280 }

287 void update_grid(const Eigen::Vector2d &center,
288 const Eigen::Matrix2d &cov,
289 std::function<float(float, float)> compute_cost_at)
290 {
291 // axis-aligned bounding box that contains the 3-sigma ellipse.
292 // Use eigenvalues so rotation is captured (cov(0,0)/(1,1) alone is not enough).
293 Eigen::SelfAdjointEigenSolver<Eigen::Matrix2d> es(cov);
294 float lam_max = static_cast<float>(es.eigenvalues().maxCoeff());
295 int r_box = static_cast<int>(std::ceil(3.0f * std::sqrt(std::max(lam_max, 0.0f))));
296
297 Eigen::Matrix2d sinv = cov.inverse();
298
299 int icx = static_cast<int>(std::round(center[0]));
300 int icy = static_cast<int>(std::round(center[1]));
301
302 // collect the cells we touched so we only call update_vertex once each.
303 std::vector<coordi> touched;
304 touched.reserve((2 * r_box + 1) * (2 * r_box + 1));
305
306 for (int dy = -r_box; dy <= r_box; dy++)
307 {
308 for (int dx = -r_box; dx <= r_box; dx++)
309 {
310 int gx = icx + dx;
311 int gy = icy + dy;
312
313 // Correct bounds check: separate gx/gy and use >= against rows/cols.
314 if (gx < 0 || gy < 0 ||
315 gx >= (int)this->cols || gy >= (int)this->rows)
316 continue;
317
318 // Mahalanobis-distance gate: drop points outside the 3σ ellipse
319 Eigen::Vector2d r;
320 r << gx - center.x(), gy - center.y();
321 double m2 = r.transpose() * sinv * r;
322 if (m2 > 9.0)
323 continue;
324
325 float computed_cost = compute_cost_at(static_cast<float>(gx),
326 static_cast<float>(gy));
327
328 coordi c{gx, gy};
329 datapoint &dp = (*this->grid)[c];
330 dp.is_obstacle = true;
331 // We *accumulate the worst* spline-penalty seen at this cell.
332 // (Field is named dist_to_spline for legacy reasons; it stores cost.)
333 // dp.dist_to_spline = std::max(dp.dist_to_spline, computed_cost);
334 touched.push_back(c);
335 }
336 }
337
338 for (const coordi &c : touched)
339 this->update_vertex(c);
340 }
