155void UnitreeLidarSDKNode::timer_callback()
156{
157 int result = lsdk_->runParse();
158 static pcl::PointCloud<PointType>::Ptr cloudOut(new pcl::PointCloud<PointType>());
159
160 // RCLCPP_INFO(this->get_logger(), "result = %d", result);
161
162 if (result == LIDAR_IMU_DATA_PACKET_TYPE)
163 {
164 LidarImuData imu;
165 lsdk_->getImuData(imu);
166
167 if (lsdk_->getImuData(imu))
168 {
169 // publish imu message
170 rclcpp::Time timestamp(imu.info.stamp.sec, imu.info.stamp.nsec);
171
172 sensor_msgs::msg::Imu imuMsg;
173 imuMsg.header.frame_id = imu_frame_;
174 imuMsg.header.stamp = timestamp;
175
176 imuMsg.orientation.x = imu.quaternion[0];
177 imuMsg.orientation.y = imu.quaternion[1];
178 imuMsg.orientation.z = imu.quaternion[2];
179 imuMsg.orientation.w = imu.quaternion[3];
180
181 imuMsg.angular_velocity.x = imu.angular_velocity[0];
182 imuMsg.angular_velocity.y = imu.angular_velocity[1];
183 imuMsg.angular_velocity.z = imu.angular_velocity[2];
184
185 imuMsg.linear_acceleration.x = imu.linear_acceleration[0];
186 imuMsg.linear_acceleration.y = imu.linear_acceleration[1];
187 imuMsg.linear_acceleration.z = imu.linear_acceleration[2];
188
189 pub_imu_->publish(imuMsg);
190
191 // publish tf from initial imu to real-time imu
192 geometry_msgs::msg::TransformStamped transformStamped;
193 transformStamped.header.stamp = this->now(); // 使用当前时间
194 transformStamped.header.frame_id = imu_frame_ + "_initial"; // 父坐标系
195 transformStamped.child_frame_id = imu_frame_; // 子坐标系
196 transformStamped.transform.translation.x = 0;
197 transformStamped.transform.translation.y = 0;
198 transformStamped.transform.translation.z = 0;
199 transformStamped.transform.rotation.x = imu.quaternion[1];
200 transformStamped.transform.rotation.y = imu.quaternion[2];
201 transformStamped.transform.rotation.z = imu.quaternion[3];
202 transformStamped.transform.rotation.w = imu.quaternion[0];
203 broadcaster_->sendTransform(transformStamped);
204
205 // publish tf from imu to lidar
206 transformStamped.header.frame_id = imu_frame_; // 父坐标系
207 transformStamped.child_frame_id = cloud_frame_; // 子坐标系
208 transformStamped.transform.translation.x = 0.007698;
209 transformStamped.transform.translation.y = 0.014655;
210 transformStamped.transform.translation.z = -0.00667;
211 transformStamped.transform.rotation.x = 0;
212 transformStamped.transform.rotation.y = 0;
213 transformStamped.transform.rotation.z = 0;
214 transformStamped.transform.rotation.w = 1;
215 broadcaster_->sendTransform(transformStamped);
216 }
217 }
218 else if (result == LIDAR_POINT_DATA_PACKET_TYPE)
219 {
220 // RCLCPP_INFO(this->get_logger(), "POINT_CLOUD");
221 PointCloudUnitree cloud;
222 if (lsdk_->getPointCloud(cloud))
223 {
224 transformUnitreeCloudToPCL(cloud, cloudOut);
225
226 rclcpp::Time timestamp(
227 static_cast<int32_t>(cloud.stamp),
228 static_cast<uint32_t>((cloud.stamp - static_cast<int32_t>(cloud.stamp)) * 1e9));
229
230 sensor_msgs::msg::PointCloud2 cloud_msg;
231 pcl::toROSMsg(*cloudOut, cloud_msg);
232 cloud_msg.header.frame_id = cloud_frame_;
233 cloud_msg.header.stamp = timestamp;
234
235 pub_cloud_->publish(cloud_msg);
236 }
237 }
